# Activity

## post/activities/

return:

	{'aid': newActivity.id, 
     'uid': newActivity.owner.id, 
     'starttime': newActivity.starttime,
     'endtime': newActivity.endtime,
     'description': newActivity.description}

	[{'aid'
    'uid'
    'starttime'
    'endtime'
	'description'}]

return:

	{'aid'
    'uid'
    'starttime'
    'endtime'
	'description'}
